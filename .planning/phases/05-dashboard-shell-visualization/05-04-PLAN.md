**src/components/CarPanel/CarFooter.tsx** — 4-column footer grid with per-corner readouts:
```typescript
import { useUIStore } from '../../stores/useUIStore'
import { fmtTemp, fmtGrip, fmtSlip, fmtEnergy } from '../../lib/formatters'
import type { LapData, Corner } from '../../lib/types'

const CORNERS: Corner[] = ['fl', 'fr', 'rl', 'rr']
const CORNER_NAMES = { fl: 'FL', fr: 'FR', rl: 'RL', rr: 'RR' }
const AXLE_LABELS = { fl: 'FRONT·L', fr: 'FRONT·R', rl: 'REAR·L', rr: 'REAR·R' }

interface CarFooterProps {
  lap: LapData | null
}

export function CarFooter({ lap }: CarFooterProps) {
  const hoveredCorner = useUIStore(s => s.hoveredCorner)
  const setHoveredCorner = useUIStore(s => s.setHoveredCorner)

  function brakeTemp(corner: Corner): number {
    if (!lap) return 0
    const e = lap[`e_tire_${corner}` as keyof LapData] as { mean: number }
    const sl = lap[`slip_angle_${corner}` as keyof LapData] as { mean: number }
    const isFront = corner.startsWith('f')
    return (isFront ? 560 : 410) + e.mean * 22 + sl.mean * 28
  }

  return (
    <div style={{
      display: 'grid',
      gridTemplateColumns: 'repeat(4, 1fr)',
      gap: 1,
      background: 'var(--rule)',
      padding: 1,
    }}>
      {CORNERS.map(c => {
        const active = hoveredCorner === c
        const temp = lap?.[`t_tread_${c}` as keyof LapData] as { mean: number; lo_95: number; hi_95: number } | undefined
        const grip = lap?.[`grip_${c}` as keyof LapData] as { mean: number; lo_95: number; hi_95: number } | undefined
        const wear = lap?.[`e_tire_${c}` as keyof LapData] as { mean: number } | undefined
        const slip = lap?.[`slip_angle_${c}` as keyof LapData] as { mean: number; lo_95: number; hi_95: number } | undefined
        const wearPct = wear ? Math.max(0, Math.min(1, wear.mean / 22)) : 0
        const wearColor = wearPct > 0.7 ? 'var(--hot)' : wearPct > 0.45 ? 'var(--warn)' : 'var(--ok)'

        return (
          <div
            key={c}
            onMouseEnter={() => setHoveredCorner(c)}
            onMouseLeave={() => setHoveredCorner(null)}
            style={{
              background: active ? 'var(--panel-header-hi)' : 'var(--panel-header)',
              padding: '6px 8px',
              borderLeft: active ? '2px solid var(--accent)' : '2px solid transparent',
              fontFamily: 'var(--mono)',
              cursor: 'default',
              minWidth: 0,
            }}
          >
            {/* Corner label */}
            <div style={{ fontSize: 8.5, fontWeight: 700, letterSpacing: 2, color: 'var(--text)', marginBottom: 4 }}>
              {CORNER_NAMES[c]} <span style={{ fontSize: 7, color: 'var(--text-muted)' }}>{AXLE_LABELS[c]}</span>
            </div>
            {/* T row */}
            <div style={{ display: 'flex', justifyContent: 'space-between', fontSize: 8, marginBottom: 2 }}>
              <span style={{ color: 'var(--text-muted)' }}>T</span>
              <span style={{ color: 'var(--text)' }}>{temp ? fmtTemp(temp) : '—'}</span>
            </div>
            {/* μ row */}
            <div style={{ display: 'flex', justifyContent: 'space-between', fontSize: 8, marginBottom: 2 }}>
              <span style={{ color: 'var(--text-muted)' }}>μ</span>
              <span style={{ color: 'var(--text)' }}>{grip ? fmtGrip(grip) : '—'}</span>
            </div>
            {/* WEAR row */}
            <div style={{ display: 'flex', justifyContent: 'space-between', fontSize: 8, marginBottom: 2 }}>
              <span style={{ color: 'var(--text-muted)' }}>WEAR</span>
              <span style={{ color: wearColor }}>{(wearPct * 100).toFixed(0)}%</span>
            </div>
            {/* α row */}
            <div style={{ display: 'flex', justifyContent: 'space-between', fontSize: 8, marginBottom: 2 }}>
              <span style={{ color: 'var(--text-muted)' }}>α</span>
              <span style={{ color: 'var(--text)' }}>{slip ? fmtSlip(slip) : '—'}</span>
            </div>
            {/* BRK row */}
            <div style={{ display: 'flex', justifyContent: 'space-between', fontSize: 8 }}>
              <span style={{ color: 'var(--text-muted)' }}>BRK</span>
              <span style={{ color: 'var(--warn)' }}>{brakeTemp(c).toFixed(0)}°C</span>
            </div>
          </div>
        )
      })}
    </div>
  )
}
```

import { useSimulationStore } from '../../stores/useSimulationStore'
import { useUIStore } from '../../stores/useUIStore'
import { PanelHeader } from '../shared/PanelHeader'
import { PanelSkeleton } from '../shared/Skeleton'
import { CarChassis } from './CarChassis'
import { CarWheel } from './CarWheel'
import { CarFooter } from './CarFooter'
import { compoundColor } from '../../lib/scales'
import type { Corner } from '../../lib/types'

const CAR_W = 400, CAR_H = 780
const TIRES = {
  fl: { cx: 82,  cy: 240, w: 46, h: 78,  isFront: true,  isLeft: true  },
  fr: { cx: 318, cy: 240, w: 46, h: 78,  isFront: true,  isLeft: false },
  rl: { cx: 76,  cy: 600, w: 54, h: 96,  isFront: false, isLeft: true  },
  rr: { cx: 324, cy: 600, w: 54, h: 96,  isFront: false, isLeft: false },
} as const

export function CarPanel() {
  const data = useSimulationStore(s => s.data)
  const pos = useUIStore(s => s.pos)
  const hoveredCorner = useUIStore(s => s.hoveredCorner)
  const setHoveredCorner = useUIStore(s => s.setHoveredCorner)

  const lapIdx = Math.min((data?.laps.length ?? 1) - 1, Math.max(0, Math.floor(pos - 1)))
  const lapNumber = lapIdx + 1
  const lap = data?.laps[lapIdx] ?? null
  const compound = data?.meta.stint.compound ?? 'MEDIUM'
  const cColor = compoundColor(compound)

  if (!data) {
    return <PanelSkeleton label="CAR — SELECT STINT AND RUN MODEL" />
  }

  return (
    <div style={{ height: '100%', display: 'grid', gridTemplateRows: '38px 1fr auto', minHeight: 0 }}>
      <PanelHeader
        title="CAR"
        subtitle="SF-24 · TOP-DOWN · INTEGRATED TELEMETRY"
        right={<span>LAP {String(lapNumber).padStart(2, '0')}</span>}
      />

      {/* SVG canvas */}
      <div style={{
        position: 'relative', overflow: 'hidden', minHeight: 0,

            </pattern>
            <radialGradient id="brake-glow">
              <stop offset="0%" stopColor="#FFB020" stopOpacity="0.9" />
              <stop offset="60%" stopColor="#FF6A00" stopOpacity="0.4" />
              <stop offset="100%" stopColor="#FF3300" stopOpacity="0" />

          </defs>
          <rect width={CAR_W} height={CAR_H} fill="url(#tech-grid)" />
          {/* Centerline */}
          <line x1={CAR_W/2} y1="10" x2={CAR_W/2} y2={CAR_H-10}
            stroke="var(--rule-strong)" strokeWidth="0.5" strokeDasharray="3 5" opacity="0.5" />
          {/* Axle reference lines */}
          <g opacity="0.35">
            <line x1="30" y1={TIRES.fl.cy} x2={CAR_W-30} y2={TIRES.fl.cy}
              stroke="var(--rule-strong)" strokeWidth="0.5" strokeDasharray="2 4" />

              stroke="var(--rule-strong)" strokeWidth="0.5" strokeDasharray="2 4" />
            <text x={CAR_W-28} y={TIRES.fl.cy-4} fill="var(--text-muted)"
              fontFamily="var(--mono)" fontSize="7" letterSpacing="1.2" textAnchor="end">FRONT AXLE</text>
            <text x={CAR_W-28} y={TIRES.rl.cy-4} fill="var(--text-muted)"
              fontFamily="var(--mono)" fontSize="7" letterSpacing="1.2" textAnchor="end">REAR AXLE</text>

          {/* Tires */}

          ))}
          {/* Compound strip */}
          <g transform="translate(140, 18)">
            <rect width="120" height="3" fill={cColor} />

              fontSize="8" textAnchor="middle" letterSpacing="2">
              {compound} · AGE {lapNumber}
            </text>

          <g fill="var(--text-muted)" fontFamily="var(--mono)" fontSize="7" letterSpacing="1">

          </g>
        </svg>
      </div>

      <CarFooter lap={lap} />
    </div>
  )
}
