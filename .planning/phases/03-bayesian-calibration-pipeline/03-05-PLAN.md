<objective>
Implement the JAX-native rewrite of Modules F (thermal ODE) and G (Arrhenius aging + mechanical wear) per D-06. This is the SINGLE MOST IMPORTANT correctness guard in Phase 3 — if the JAX model diverges from the NumPy production path, Stage 4's posterior is wrong. RESEARCH.md Pattern 2 specifies the exact implementation shape. The production NumPy code in `packages/core/src/f1_core/physics/` stays untouched.

Key design decisions from CONTEXT.md D-06: since Stages 1-3 pre-compute `T_tread` and `P_slide` trajectories and pass them as FIXED inputs to Stage 4, the JAX scan only needs to step the scalar `mu_0` and per-tire `d_tread` — a minimal 3-parameter likelihood. This means `simulate_mu_0` is much simpler than full thermal integration, and extremely fast per MCMC sample.

Purpose: Without the parity test, a subtle JAX translation bug (e.g., float32 vs float64, incorrect scan order, sign flip on Arrhenius exponent) would silently corrupt Stage 4 results.

Output: Pure-JAX `jax_model.py` with `simulate_mu_0`, `log_likelihood_f_g`, and optional full `thermal_scan`; parity test file comparing JAX output to NumPy module_g.degradation_step output to 1e-6 absolute tolerance.

    ```python
    """JAX-native Modules F + G rewrite for Stage 4 MCMC (CONTEXT D-06).

    CRITICAL: this file is a PARITY COPY of
      - packages/core/src/f1_core/physics/module_f.thermal_step  (full thermal ODE)
      - packages/core/src/f1_core/physics/module_g.degradation_step  (Arrhenius + wear)
    The NumPy production path in packages/core/ is untouched (D-06).

    Pitfall 1 (Pitfall 1): jax_enable_x64 MUST be set before any tracing.
    Pitfall 4 (from module_g): Arrhenius exponent is clamped at ±20 to prevent overflow.

    Design: because Stages 1-3 pre-compute T_tread_traj and P_slide_traj as FIXED inputs,
    Stage 4 only steps mu_0 (scalar) and d_tread ((4,) per-tire). This is a minimal
    3-parameter scan — ideal for jax.lax.scan and extremely fast per MCMC sample.
    """
    from __future__ import annotations

    import jax
    jax.config.update("jax_enable_x64", True)   # Pitfall 1 — must be at module load

    import jax.numpy as jnp
    from jax import lax

    # Parity constants (match packages/core/src/f1_core/physics/constants.py + module_g)
    DT: float = 0.25                      # DT_THERMAL
    T_REF_AGING: float = 80.0             # model_spec §G.2
    ARRHENIUS_EXP_CLAMP: float = 20.0     # module_g.ARRHENIUS_EXP_CLAMP


    def _simulate_mu_0_step(carry, step_inputs):
        """One scan step: carry = (mu_0, d_tread); step_inputs = (t_tread_row, p_slide_row).

        Closed over (beta_therm, T_act, k_wear) via the outer simulate_mu_0 wrapper.
        """
        (mu_0, d_tread, beta_therm, T_act, k_wear) = carry
        t_tread_row, p_slide_row = step_inputs

        # §G.2 — Arrhenius aging on MEAN tread temperature (module_g line-for-line)
        t_tread_mean = jnp.mean(t_tread_row)
        arg = jnp.clip((t_tread_mean - T_REF_AGING) / T_act,
                       -ARRHENIUS_EXP_CLAMP, ARRHENIUS_EXP_CLAMP)
        d_mu_0_dt = -beta_therm * mu_0 * jnp.exp(arg)
        mu_0_new = jnp.maximum(mu_0 + DT * d_mu_0_dt, 0.0)

        # §G.3 — per-tire wear
        d_tread_new = jnp.maximum(d_tread - DT * k_wear * p_slide_row, 0.0)

        new_carry = (mu_0_new, d_tread_new, beta_therm, T_act, k_wear)
        return new_carry, mu_0_new   # emit mu_0_new at each step


    def simulate_mu_0(
        beta_therm: jnp.ndarray | float,
        T_act: jnp.ndarray | float,
        k_wear: jnp.ndarray | float,
        *,
        t_tread_traj: jnp.ndarray,   # (N, 4) fixed from Stage 3 forward pass
        p_slide_traj: jnp.ndarray,   # (N, 4) fixed from Stage 3 forward pass
        mu_0_init: float,
        d_tread_init: jnp.ndarray,   # (4,)
    ) -> jnp.ndarray:
        """Return (N,) mu_0 trajectory by stepping only aging + wear (T_tread fixed)."""
        beta_therm = jnp.asarray(beta_therm, dtype=jnp.float64)
        T_act = jnp.asarray(T_act, dtype=jnp.float64)
        k_wear = jnp.asarray(k_wear, dtype=jnp.float64)
        t_tread_traj = jnp.asarray(t_tread_traj, dtype=jnp.float64)
        p_slide_traj = jnp.asarray(p_slide_traj, dtype=jnp.float64)
        d_tread_init = jnp.asarray(d_tread_init, dtype=jnp.float64)
        mu_0_init_j = jnp.asarray(mu_0_init, dtype=jnp.float64)

        init_carry = (mu_0_init_j, d_tread_init, beta_therm, T_act, k_wear)
        _, mu_0_traj = lax.scan(_simulate_mu_0_step, init_carry, (t_tread_traj, p_slide_traj))
        return mu_0_traj


    def _thermal_step_jax(carry, step_inputs):
        """Parity copy of module_f.thermal_step for optional full thermal scan.

        Not used by Stage 4 (which takes t_tread_traj as fixed), but available for
        sanity checks or v2 experiments.
        """
        (t_tread, t_carc, t_gas, C_tread, C_carc, C_gas,
         R_tc, R_cg, h_0, h_1, alpha_p,
         A_tread, A_carc, H_CARC) = carry
        v, t_air, p_total = step_inputs

        # §F.5
        v_safe = jnp.maximum(v, 0.0)
        h_air = h_0 + h_1 * jnp.sqrt(v_safe)
        # §F.1
        q_heat = alpha_p * p_total
        q_conv_tread = h_air * A_tread * (t_tread - t_air)
        q_tc = (t_tread - t_carc) / R_tc
        dT_tread = (q_heat - q_conv_tread - q_tc) / C_tread
        # §F.2
        q_conv_carc = H_CARC * A_carc * (t_carc - t_air)
        q_cg = (t_carc - t_gas) / R_cg
        dT_carc = (q_tc - q_conv_carc - q_cg) / C_carc
        # §F.3
        dT_gas = q_cg / C_gas
        # §F.7
        t_tread_new = t_tread + DT * dT_tread
        t_carc_new = t_carc + DT * dT_carc
        t_gas_new = t_gas + DT * dT_gas

        new_carry = (t_tread_new, t_carc_new, t_gas_new, C_tread, C_carc, C_gas,
                     R_tc, R_cg, h_0, h_1, alpha_p, A_tread, A_carc, H_CARC)
        return new_carry, t_tread_new


    def thermal_scan(
        *,
        thermal_params: tuple,        # (C_tread, C_carc, C_gas, R_tc, R_cg, h_0, h_1, alpha_p)
        v_traj: jnp.ndarray,           # (N,)
        t_air: float,
        p_total_traj: jnp.ndarray,    # (N, 4)
        t0: float,                     # initial tread/carc/gas (t_track + delta_T_blanket)
        A_tread: jnp.ndarray,          # (4,) per-tire convective areas front/rear
        A_carc: jnp.ndarray,           # (4,)
        H_CARC: float = 5.0,
    ) -> jnp.ndarray:
        """Full thermal scan — optional; not in the Stage 4 likelihood path."""
        C_tread, C_carc, C_gas, R_tc, R_cg, h_0, h_1, alpha_p = thermal_params
        init = (
            jnp.full((4,), t0, dtype=jnp.float64),
            jnp.full((4,), t0, dtype=jnp.float64),
            jnp.full((4,), t0, dtype=jnp.float64),
            jnp.asarray(C_tread, jnp.float64), jnp.asarray(C_carc, jnp.float64),
            jnp.asarray(C_gas, jnp.float64), jnp.asarray(R_tc, jnp.float64),
            jnp.asarray(R_cg, jnp.float64), jnp.asarray(h_0, jnp.float64),
            jnp.asarray(h_1, jnp.float64), jnp.asarray(alpha_p, jnp.float64),
            jnp.asarray(A_tread, jnp.float64), jnp.asarray(A_carc, jnp.float64),
            jnp.asarray(H_CARC, jnp.float64),
        )
        t_air_arr = jnp.full_like(v_traj, t_air, dtype=jnp.float64)
        _, t_tread_traj = lax.scan(
            _thermal_step_jax, init,
            (v_traj.astype(jnp.float64), t_air_arr, p_total_traj.astype(jnp.float64)),
        )
        return t_tread_traj


    def log_likelihood_f_g(
        beta_therm, T_act, k_wear, sigma_obs,
        *,
        obs_lap_times: jnp.ndarray,     # (L,) observed per-lap times (s)
        lap_boundary_idx: jnp.ndarray,  # (L+1,) indices into (N,) mu_0 trajectory at lap starts
        t_tread_traj: jnp.ndarray,      # (N, 4)
        p_slide_traj: jnp.ndarray,      # (N, 4)
        mu_0_init: float,
        d_tread_init: jnp.ndarray,      # (4,)
        t_lap_ref: float,
    ) -> jnp.ndarray:
        """Per-lap Gaussian log-likelihood on Δt_lap predictions (model_spec §G.4)."""
        mu_0_traj = simulate_mu_0(
            beta_therm, T_act, k_wear,
            t_tread_traj=t_tread_traj, p_slide_traj=p_slide_traj,
            mu_0_init=mu_0_init, d_tread_init=d_tread_init,
        )
        # Take μ_0 at end of each lap — lap_boundary_idx has L+1 entries
        end_idx = jnp.asarray(lap_boundary_idx[1:] - 1, dtype=jnp.int32)
        mu_0_end = mu_0_traj[end_idx]
        # §G.4 lap-time penalty: Δt_lap ≈ (t_ref/2) · (μ₀_fresh − μ₀(t)) / μ₀_fresh
        pred_lap_times = t_lap_ref + 0.5 * t_lap_ref * (mu_0_init - mu_0_end) / mu_0_init
        resid = (obs_lap_times - pred_lap_times) / sigma_obs
        return jnp.sum(-0.5 * resid ** 2)


    __all__ = [
        "DT", "T_REF_AGING", "ARRHENIUS_EXP_CLAMP",
        "simulate_mu_0", "thermal_scan", "log_likelihood_f_g",
    ]
    ```
