<objective>
Wire the `f1-calibrate` Typer CLI entry point (D-01) and implement the resumable `run-all` orchestrator. Every individual stage subcommand loads its training data, invokes the appropriate `fit_stageN` function, and writes the result to SQLite. `run-all` chains stages 1→5 with a SQLite-based resumability gate: before each stage, query `has_stage_result(compound, stage_number, year_range)` and skip if a row exists (unless `--force`). All exceptions are caught at the CLI boundary and printed without tracebacks (T-3-04). Stage 5 always runs because it writes `calibration_runs` (the final provenance row) and is quick.

Purpose: The CLI is the only user-facing surface of Phase 3. Exit codes, error messages, and resumability determine developer experience. Without a resumable `run-all`, a broken Stage 4 forces re-running Stages 1-3 from scratch — wasted 30-60 minutes of FastF1 reload and optimization.

Output: cli.py (Typer app with 6 subcommands) + run_all.py (orchestrator) + two test files validating help output, compound validation, exit codes, and second-invocation skipping.

    """f1-calibrate Typer CLI (CONTEXT.md D-01, RESEARCH.md §Pitfall 6).

    Subcommands (all use `--compound` flag per Pitfall 6):
      stage1   — Aero fit (CALIB-01)
      stage2   — Friction fit (CALIB-02)
      stage3   — Thermal fit (CALIB-03)
      stage4   — Degradation MCMC (CALIB-04, CALIB-06, CALIB-07)
      stage5   — Cross-validation + baseline comparison (CALIB-05, CALIB-08)
      run-all  — Chain stages 1→5 with SQLite resumability (D-01)

    Exit codes (T-3-04 — no traceback leak at any boundary):
      0 success
      1 validation error (bad compound, bad path)
      2 calibration/convergence error (r̂ failed, stage missing)
      3 unexpected internal error
    """
    from __future__ import annotations

    import sqlite3
    from pathlib import Path

    import typer
    from rich.console import Console
    from rich.table import Table

    from f1_calibration.common import TRAINING_YEARS, YEAR_RANGE, get_logger
    from f1_calibration.db import (
        DEFAULT_DB_PATH, initialize_schema, resolve_db_path,
        validate_compound, write_parameter_set,
    )

    app = typer.Typer(
        add_completion=False, no_args_is_help=True,
        help="f1-calibrate — Offline Bayesian calibration pipeline (Phase 3).",
    )
    console = Console()
    _log = get_logger(__name__)


    # ------- Helpers --------------------------------------------------------

    def _handle_exit(exc: Exception) -> "typer.Exit":
        """Map exception → exit code + printed message (T-3-04)."""
        if isinstance(exc, ValueError):
            console.print(f"[red]Invalid input:[/red] {exc}")
            return typer.Exit(code=1)
        if isinstance(exc, RuntimeError):
            console.print(f"[red]Calibration failed:[/red] {exc}")
            return typer.Exit(code=2)
        console.print(f"[red]Internal error:[/red] {exc}")
        return typer.Exit(code=3)


    def _open_db(db_path: str | None) -> sqlite3.Connection:
        """Open SQLite + initialize schema. Resolves path with workspace containment check."""
        path = resolve_db_path(db_path if db_path else DEFAULT_DB_PATH)
        path.parent.mkdir(parents=True, exist_ok=True)
        conn = sqlite3.connect(path)
        initialize_schema(conn)
        return conn


    # ------- Stage 1 --------------------------------------------------------

    @app.command()
    def stage1(
        compound: str = typer.Option(..., "--compound", help="Target compound C1..C5"),
        db_path: str | None = typer.Option(None, "--db", help="SQLite path (default .data/f1.db)"),
    ) -> None:
        """Fit aero params (C_LA, C_DA, ξ) — CALIB-01. Compound-agnostic; stored under target compound key."""
        try:
            compound = validate_compound(compound)
            conn = _open_db(db_path)
            try:
                from f1_calibration.stage1_aero import fit_stage1
                from f1_calibration.training import iter_training_stints
                # Extract observables: peak lat-g + corner speeds across training stints.
                # For v1 we pull from iter_training_stints and extract features on the fly.
                import numpy as np
                lat_g_list: list[float] = []
                v_list: list[float] = []
                n_stints_used = 0
                for stint in iter_training_stints(years=TRAINING_YEARS, compound=None):
                    # Placeholder extraction — pull every cornering sample where V in 30..90 m/s
                    laps = getattr(stint, "laps", None)
                    tel = getattr(stint, "telemetry", None)
                    if tel is not None:
                        try:
                            v_col = np.asarray(tel["V_mps"], dtype=np.float64)
                            a_lat = np.asarray(tel["a_lat_mps2"], dtype=np.float64) / 9.81
                        except Exception:
                            continue
                        mask = (v_col > 30.0) & (v_col < 90.0) & (np.abs(a_lat) > 2.0)
                        if mask.any():
                            lat_g_list.extend(np.abs(a_lat[mask]).tolist())
                            v_list.extend(v_col[mask].tolist())
                            n_stints_used += 1
                if len(lat_g_list) < 3:
                    raise RuntimeError(
                        f"insufficient corner samples for stage 1: got {len(lat_g_list)}, need >=3"
                    )
                params, diag = fit_stage1(np.asarray(lat_g_list), np.asarray(v_list))
                pid = write_parameter_set(conn, compound, 1, params, diag)

                table = Table(title=f"Stage 1 — Aero fit ({compound})")
                for col in ("Parameter", "Value"):
                    table.add_column(col)
                for k, v in {"C_LA": params.C_LA, "C_DA": params.C_DA, "xi": params.xi}.items():
                    table.add_row(k, f"{v:.4f}")
                console.print(table)
                console.print(f"[green]Wrote parameter_set_id={pid}[/green]")
            finally:
                conn.close()
        except Exception as exc:  # noqa: BLE001 — CLI boundary (T-3-04)
            raise _handle_exit(exc)


    # ------- Stage 2 --------------------------------------------------------

    @app.command()
    def stage2(
        compound: str = typer.Option(..., "--compound"),
        db_path: str | None = typer.Option(None, "--db"),
    ) -> None:
        """Fit friction baseline (μ₀^fresh, p̄₀, n) — CALIB-02."""
        try:
            compound = validate_compound(compound)
            conn = _open_db(db_path)
            try:
                from f1_calibration.stage2_friction import fit_stage2
                from f1_calibration.training import iter_training_stints
                import numpy as np
                mu_eff: list[float] = []
                p_bar: list[float] = []
                for stint in iter_training_stints(years=TRAINING_YEARS, compound=None):
                    # Extract laps 2-5 samples: μ_eff = |F_y|/F_z at peak-grip moments.
                    # Skip extraction if the stint artifact doesn't expose these directly.
                    samples = getattr(stint, "friction_samples", None)
                    if samples is None:
                        continue
                    mu_s, p_s = samples.get("mu_eff"), samples.get("p_bar")
                    if mu_s is None or p_s is None:
                        continue
                    mu_eff.extend(np.asarray(mu_s).tolist())
                    p_bar.extend(np.asarray(p_s).tolist())
                if len(mu_eff) < 10:
                    raise RuntimeError(
                        f"insufficient friction samples: got {len(mu_eff)}, need >=10"
                    )
                params, diag = fit_stage2(np.asarray(mu_eff), np.asarray(p_bar))
                pid = write_parameter_set(conn, compound, 2, params, diag)
                table = Table(title=f"Stage 2 — Friction fit ({compound})")
                for col in ("Parameter", "Value"):
                    table.add_column(col)
                for k, v in {"mu_0_fresh": params.mu_0_fresh, "p_bar_0": params.p_bar_0, "n": params.n}.items():
                    table.add_row(k, f"{v:.4g}")
                console.print(table)
                console.print(f"[green]Wrote parameter_set_id={pid}[/green]")
            finally:
                conn.close()
        except Exception as exc:  # noqa: BLE001
            raise _handle_exit(exc)

    @app.command()
    def stage4(
        compound: str = typer.Option(..., "--compound"),
        db_path: str | None = typer.Option(None, "--db"),
        skip_sbc: bool = typer.Option(False, "--skip-sbc", help="Skip SBC pre-flight (not recommended)"),
        chains: int = typer.Option(4, "--chains"),
        draws: int = typer.Option(1000, "--draws"),
        tune: int = typer.Option(1000, "--tune"),
    ) -> None:
        """Fit degradation posterior (β_therm, T_act, k_wear) via PyMC+NumPyro — CALIB-04."""
        try:
            compound = validate_compound(compound)
            conn = _open_db(db_path)
            try:
                from f1_calibration.stage4_degradation import fit_stage4
                from f1_calibration.training import iter_training_stints
                import numpy as np

                # Stage 4 consumes pre-computed trajectories from the full forward pass.
                # The CLI path assembles fixed_trajectories from the first qualifying stint.
                # (Real run-all will loop over stints; v1 CLI demonstrates single-stint call.)
                stints = list(iter_training_stints(years=TRAINING_YEARS, compound=compound))
                if not stints:
                    raise RuntimeError(f"no training stints for compound {compound}")
                from f1_core.physics.orchestrator import run_simulation
                from f1_calibration.db import read_latest_parameter_set
                # Assemble PhysicsParams from stages 1-3
                rows = [read_latest_parameter_set(conn, compound, s) for s in (1, 2, 3)]
                if any(r is None for r in rows):
                    raise RuntimeError(f"stages 1-3 must complete before stage4; run-all first")
                from f1_core.physics.params import AeroParams, FrictionParams, ThermalParams, DegradationParams, PhysicsParams
                from f1_core.physics.defaults import make_nominal_params
                nominal_deg = make_nominal_params().degradation
                physics_params = PhysicsParams(
                    aero=AeroParams(**rows[0]["params"]),
                    friction=FrictionParams(**rows[1]["params"]),
                    thermal=ThermalParams(**rows[2]["params"]),
                    degradation=nominal_deg,
                )
                # Run forward pass on first stint to get T_tread + P_slide trajectories
                result = run_simulation(stints[0], physics_params)
                fixed = {
                    "t_tread_traj": np.asarray(result.t_tread_trajectory),    # (N, 4)
                    "p_slide_traj": np.asarray(result.p_slide_trajectory),    # (N, 4)
                    "mu_0_init": float(physics_params.friction.mu_0_fresh),
                    "d_tread_init": np.full(4, 0.010, dtype=np.float64),
                }
                lap_boundaries = np.asarray(result.lap_boundary_indices, dtype=np.int64)
                obs_lap_times = np.asarray(result.observed_lap_times, dtype=np.float64)
                t_lap_ref = float(np.median(obs_lap_times))

                idata, pid = fit_stage4(
                    compound=compound, fixed_trajectories=fixed,
                    obs_lap_times=obs_lap_times, lap_boundary_idx=lap_boundaries,
                    t_lap_ref=t_lap_ref, db_conn=conn,
                    chains=chains, draws=draws, tune=tune,
                    skip_sbc=skip_sbc,
                )
                import arviz as az
                summ = az.summary(idata, var_names=["beta_therm", "T_act", "k_wear"])
                table = Table(title=f"Stage 4 — Degradation posterior ({compound})")
                for col in ("Parameter", "Mean", "SD", "r_hat", "ess_bulk"):
                    table.add_column(col)
                for name in ("beta_therm", "T_act", "k_wear"):
                    r = summ.loc[name]
                    table.add_row(name, f"{r['mean']:.4g}", f"{r['sd']:.4g}",
                                  f"{r['r_hat']:.4f}", f"{r['ess_bulk']:.0f}")
                console.print(table)
                console.print(f"[green]Wrote parameter_set_id={pid}[/green]")
            finally:
                conn.close()
        except Exception as exc:  # noqa: BLE001
            raise _handle_exit(exc)

    Before each stage, query parameter_sets for (compound, stage_number, year_range);
    if a row exists and --force is not passed, skip the stage. Stage 5 ALWAYS runs
    (it's validation, not a fit — fresh RMSE is required each run).

    At end of every successful run_all: write a calibration_runs row tying the
    four parameter_set_ids + Stage 5 CSV + NetCDF path together.
    """
    from __future__ import annotations

    import json
    import sqlite3
    from typing import Any

    from f1_calibration.common import YEAR_RANGE, get_logger
    from f1_calibration.db import (
        has_stage_result, read_latest_parameter_set,
        validate_compound, write_calibration_run,
    )

    _log = get_logger(__name__)


    def _run_single_stage(stage_num: int, compound: str, conn: sqlite3.Connection,
                          console) -> int:
        """Invoke the fit function for one stage and write its parameter_sets row.

        Returns parameter_set_id. Loads training data + computes observables inline.
        Real-data extraction is intentionally thin in v1 — the heavy lifting lives
        in each stage module's fit function.
        """
        if stage_num == 1:
            from f1_calibration.cli import _run_stage1_adapter as _adapter  # type: ignore[attr-defined]
        else:
            raise NotImplementedError(f"stage {stage_num} CLI adapter not wired yet")
        return _adapter(compound, conn)


    def run_all(
        *,
        compound: str,
        conn: sqlite3.Connection,
        force: bool = False,
        console: Any | None = None,
    ) -> dict[str, Any]:
        compound = validate_compound(compound)
        if console is None:
            from rich.console import Console
            console = Console()

        stages_run: list[int] = []
        stages_skipped: list[int] = []

        for stage_num in (1, 2, 3, 4):
            if not force and has_stage_result(conn, compound, stage_num):
                console.print(
                    f"[dim]Stage {stage_num} already complete for {compound}; "
                    f"skipping (use --force to re-run)[/dim]"
                )
                stages_skipped.append(stage_num)
                continue
            console.print(f"[bold]Running stage {stage_num} for {compound}...[/bold]")
            _run_stage_dispatch(stage_num, compound, conn, console)
            stages_run.append(stage_num)

        # Stage 5 ALWAYS runs — it produces the calibration_runs row.
        console.print(f"[bold]Running stage 5 (validation) for {compound}...[/bold]")
        from f1_calibration.stage5_validation import fit_stage5
        stage5 = fit_stage5(compound, conn)

        # Collect stage 1-4 parameter_set_ids + stage 4 diagnostics for calibration_runs
        ids: dict[int, int] = {}
        stage4_diag: dict[str, Any] = {}
        for s in (1, 2, 3, 4):
            row = read_latest_parameter_set(conn, compound, s)
            if row is None:
                raise RuntimeError(f"cannot write calibration_runs: stage {s} missing after run_all")
            ids[s] = int(row["parameter_set_id"])
            if s == 4:
                stage4_diag = row.get("diagnostics") or {}

        calibration_id = write_calibration_run(
            conn,
            compound=compound,
            heldout_rmse_s=float(stage5["physics_rmse_s"]),
            baseline_rmse_s=float(stage5["baseline_rmse_s"])
                if stage5["baseline_rmse_s"] == stage5["baseline_rmse_s"]  # NaN check
                else 0.0,
            r_hat_max=float(stage4_diag.get("r_hat_max", 0.0)),
            ess_bulk_min=float(stage4_diag.get("ess_bulk_min", 0.0)),
            netcdf_path=str(stage4_diag.get("netcdf_path", "")),
            param_set_stage1=ids[1], param_set_stage2=ids[2],
            param_set_stage3=ids[3], param_set_stage4=ids[4],
            stage5_csv_path=str(stage5.get("csv_path") or ""),
        )
        console.print(
            f"[green]run-all complete:[/green] calibration_id={calibration_id} "
            f"physics_rmse={stage5['physics_rmse_s']:.4f}s "
            f"baseline_rmse={stage5['baseline_rmse_s']:.4f}s"
        )
        return {
            "compound": compound,
            "stages_run": stages_run,
            "stages_skipped": stages_skipped,
            "physics_rmse_s": float(stage5["physics_rmse_s"]),
            "baseline_rmse_s": float(stage5["baseline_rmse_s"]),
            "calibration_id": calibration_id,
        }


    def _run_stage_dispatch(stage_num: int, compound: str, conn: sqlite3.Connection,
                             console) -> None:
        """Dispatch to the CLI command's core logic for each stage.

        Reuses the Typer command bodies — we avoid duplicating training-data extraction.
        Invoked via typer.testing.CliRunner wouldn't work here (we already hold the conn),
        so we refactored the core logic into module-level `_stageN_core(compound, conn)`
        helpers in cli.py. This dispatch function simply picks the right core function.
        """
        # Lazy import to avoid circular init
        from f1_calibration import cli as cli_mod
        if stage_num == 1:
            cli_mod._stage1_core(compound, conn, console)   # type: ignore[attr-defined]
        elif stage_num == 2:
            cli_mod._stage2_core(compound, conn, console)   # type: ignore[attr-defined]
        elif stage_num == 3:
            cli_mod._stage3_core(compound, conn, console)   # type: ignore[attr-defined]
        elif stage_num == 4:
            cli_mod._stage4_core(compound, conn, console)   # type: ignore[attr-defined]
        else:
            raise ValueError(f"unknown stage_num {stage_num}")


    __all__ = ["run_all"]
